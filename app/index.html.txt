<script>
const SUPABASE_URL = "https://rzgrdamzgeyyotazrapz.supabase.co";
const SUPABASE_ANON = "sb_publishable_Po8DdRAuP2wUijaNPMOEXQ_1Mf5gE0s";
const AI_URL = "https://andrewcj999-vibeshift-ai.hf.space";

const supa = supabase.createClient(SUPABASE_URL, SUPABASE_ANON);

let isLogin = false;
const authDiv = document.getElementById('auth');
const appDiv = document.getElementById('app');

document.getElementById('toggle').onclick = () => {
  isLogin = !isLogin;
  document.getElementById('auth-title').textContent = isLogin ? 'Login' : 'Sign Up';
  document.getElementById('auth-btn').textContent = isLogin ? 'Login' : 'Create Account';
  document.getElementById('toggle').textContent = isLogin ? 'Need account? Sign up' : 'Already have account? Login';
};

document.getElementById('auth-btn').onclick = async () => {
  const email = document.getElementById('email').value;
  const pass = document.getElementById('password').value;
  const btn = document.getElementById('auth-btn');
  btn.disabled = true;
  
  const fn = isLogin ? supa.auth.signInWithPassword : supa.auth.signUp;
  const {error} = await fn({email, password: pass});
  
  if(error) alert(error.message);
  else checkAuth();
  btn.disabled = false;
};

document.getElementById('logout').onclick = async () => {
  await supa.auth.signOut();
  location.reload();
};

async function checkAuth() {
  const {data} = await supa.auth.getUser();
  if(data.user) {
    authDiv.classList.add('hidden');
    appDiv.classList.remove('hidden');
    loadStyles();
  }
}

document.getElementById('analyze').onclick = async () => {
  const url = document.getElementById('yturl').value;
  if(!url) return alert('Paste YouTube URL');
  
  const btn = document.getElementById('analyze');
  btn.disabled = true;
  btn.textContent = 'Analyzing...';
  
  try {
    const res = await fetch(AI_URL + '/analyze', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({youtube_url: url})
    });
    const data = await res.json();
    
    document.getElementById('title').textContent = data.title;
    document.getElementById('vibe').textContent = data.vibe;
    document.getElementById('thumb').src = data.thumbnail;
    
    const colorsDiv = document.getElementById('colors');
    colorsDiv.innerHTML = '';
    data.colors.forEach(c => {
      const d = document.createElement('div');
      d.className = 'color';
      d.style.background = c;
      d.title = c;
      colorsDiv.appendChild(d);
    });
    
    document.getElementById('result').classList.remove('hidden');
    
    const {data: {user}} = await supa.auth.getUser();
    await supa.from('styles').insert({
      user_id: user.id,
      name: data.title,
      youtube_url: url,
      vibe: data.vibe,
      colors: data.colors
    });
    
    loadStyles();
    document.getElementById('yturl').value = '';
  } catch(e) {
    alert('Error: ' + e.message);
  }
  
  btn.disabled = false;
  btn.textContent = 'Analyze Vibe';
};

async function loadStyles() {
  const {data} = await supa.from('styles').select('*').order('created_at', {ascending:false});
  const div = document.getElementById('styles');
  if(!data || data.length === 0) {
    div.innerHTML = '<p style="color:#666">No styles yet. Analyze one above.</p>';
    return;
  }
  div.innerHTML = data.map(s => `
    <div class="style-item">
      <strong>${s.name}</strong>
      <p style="color:#888;font-size:14px;margin:4px 0">${s.vibe}</p>
      <div style="display:flex;gap:4px;margin-top:8px">
        ${s.colors.map(c => `<div style="width:24px;height:24px;background:${c};border-radius:4px"></div>`).join('')}
      </div>
    </div>
  `).join('');
}

checkAuth();
</script>
